Defines concat kernel for ArrayRef

Example:

use arrow_array::{ArrayRef, StringArray};
use arrow_select::concat::concat;

let arr = concat(&[
    &StringArray::from(vec!["hello", "world"]),
    &StringArray::from(vec!["!"]),
]).unwrap();
assert_eq!(arr.len(), 3);
